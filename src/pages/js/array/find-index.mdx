## js

```js
const words = ["a", "b", "c"];
// 주어진 판별 함수를 만족하는 배열의 첫 번째 요소에 대한 인덱스를 반환
const findIndex = words.findIndex(function (word) {
  return word === "a";
});
console.log(findIndex);
```
